\section{Function Composition}

\begin{definition}[Composite function]
    Suppose $f$ and $g$ are two functions. The \textbf{composite} if $g$ with $f$, denoted $f \circ g$ is defined by the formula $(g \circ f)(x) = g(f(x))$, provided $x$ is an element of the domain of $f$ and $f(x)$ is an element of the domain of $g$
\end{definition}

\begin{theorem}[Properties of Function Composition]
    Suppose $f$, $g$, and $h$ are functions.
    \begin{itemize}
        \item $h \circ (g \circ f) = (h \circ g) \circ f$, provided the composite functions are defined.
        \item If $I$ is defined as $I(x) = x$ for all real numbers $x$, then $I \circ f = f \circ I = f$
    \end{itemize}
\end{theorem}
