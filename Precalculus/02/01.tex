\section{Linear Functions}

\begin{definition}[Slope]
    The \textbf{slope} $m$ of the line containing the points $P(x_0, y_0)$ and $Q(x_1,y_1)$ is:
    \[m  = \frac{y_1 - y_0}{x_1 - x_0} = \frac{\Delta y}{\Delta x}\]
    provided $x_1 \neq x_0$.
\end{definition}

\begin{definition}[Point-slope form]
    The \textbf{point-slope form} of the line with slope $m$ containing the point $(x_0, y_0)$ is the equation $y - y_0 = m (x - x_0)$.
\end{definition}

\begin{definition}[Slope-intercept form]
    The \textbf{slope-intercept form} of the line with slope $m$ and $y$-intercept $(0,b)$ is the equation $y = mx + b$.
\end{definition}

\begin{definition}[Linear function]
    A \textbf{linear function} is a function of the form
    \[f(x) = mx + b\]
    where $m$ and $b$ are real numbers with $m \neq 0$. The domain of a linear function is $(-\infty,\infty)$.
\end{definition}

\begin{definition}[Constant function]
    A \textbf{constant function} is a function of the form
    \[f(x) = b\],
    where $b$ is a real number. The domain of a constant function is $(-\infty, \infty)$.
\end{definition}

\begin{definition}[Average rate of change]
    Let $f$ be a function defined on the interval $[a, b]$. The \textbf{average rate of change} of $f$ over $[a,b]$ is defined as:
    \[\frac{\Delta f}{\Delta x} = \frac{f(b) - f(a)}{b - a}\].
    The line that connects $(a, f(a))$ with $(b, f(b))$ is called the \textbf{secant line}.
\end{definition}
