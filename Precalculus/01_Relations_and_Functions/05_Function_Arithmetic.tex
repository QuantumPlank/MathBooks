\section{Function Arithmetic}

\begin{definition}[Function Arithmetic]
      Suppose that $f$ and $g$ are functions and $x$ is in both the domain of $f$ and the domain of $g$.
      \begin{itemize}
            \item The \textbf{sum} of $f$ and $g$, denoted $f+g$, is the function defined by the formula
                  \[(f+g)(x) = f(x) + g(x)\]
            \item The \textbf{difference} of $f$ and $g$, denoted $f-g$, is the function defined by the formula
                  \[(f-g)(x) = f(x) - g(x)\]
            \item The \textbf{product} of $f$ and $g$, denoted $fg$, is the function defined by the formula
                  \[(fg)(x) = f(x)g(x)\]
            \item The \textbf{quotient} of $f$ and $g$, denoted $\frac{f}{g}$, is the function defined by the formula
                  \[(\frac{f}{g})(x) = \frac{f(x)}{g(x)}\]
      \end{itemize}
\end{definition}

\begin{definition}[Difference Quotient]
      Given a function $f$, the \textbf{difference quotient} of $f$ is the expression
      \[\frac{f(x+h)-f(x)}{h}\]
\end{definition}
