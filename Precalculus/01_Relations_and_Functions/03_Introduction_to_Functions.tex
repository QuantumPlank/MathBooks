\section{Introduction to Functions}

\begin{definition}[Function]
    A relation in which each \textit{x-coordinate} is matched with only one \textit{y-coordinate} is said to describe $y$ as a \textbf{function} of $x$
\end{definition}

\begin{definition}[Vertical Line Test]
    A set of points in the plane represents a $y$ as a function of $x$ if and only if no two points lie on the same vertical line.
\end{definition}

\begin{definition}[Domain and Range of a function]
    Suppose that $F$ is a relation which describes $y$ as a function of $x$
    \begin{itemize}
        \item The set of the \textit{x-coordinates} of the points in $F$ is called the \textbf{domain} of $F$
        \item The set of the \textit{y-coordinates} of the points in $F$ is called the \textbf{range} of $F$
    \end{itemize}
\end{definition}
