\section{Properties of Logarithms}

\begin{theorem}[Inverse Properties of Exponential and Logarithmic Functions]
    Let $b>0$, $b \neq 1$.
    \begin{itemize}
        \item $b^a = c$ if and only if $\log_{b}(c)$.
        \item $\log_{b}(b^x)=x$ for all $x$ and $b^{\log_{b}(x)}=x$ for all $x > 0$.
    \end{itemize}
\end{theorem}

\begin{theorem}[One-to-one Properties of Exponential and Logarithmic Functions]
    Let $f(x) = b^x$ and $g(x)=log_{b}(x)$ where $b>0$, $b \neq 1$. Then $f$ and $g$ are one-to-one and
    \begin{itemize}
        \item $b^u=b^w$ if and only if $u=w$ for all real numbers $u$ and $w$.
        \item $\log_{b}(u) = \log_{b}(w)$ if and only if $u=w$ for all real numbers $u>0$ and $w>0$.
    \end{itemize}
\end{theorem}

\begin{theorem}[Algebraic Properties of Exponential Functions]
    Let $f(x)=b^x$ be an exponential function $(b>0, b\neq 1)$ and let $u$ and $w$ be real numbers.
    \begin{itemize}
        \item \textbf{Product Rule}: $f(u+w) = f(u)f(w)$. In other words, $b^{u+w}=b^u b^w$.
        \item \textbf{Quotient Rule}: $f(u-w) = \frac{f(u)}{f(w)}$. In other words, $b^{u-w}=\frac{b^u}{b^w}$.
        \item \textbf{Power Rule}: $f(u)^w = f(uw)$. In other words, $(b^u)^w=b^{uw}$.
    \end{itemize}
\end{theorem}

\begin{theorem}[Algebraic Properties of Logarithmic Functions]
    Let $g(x)=\log_{b}(x)$ be a Logarithmic function $(b>0, b\neq 1)$ and let $u>0$ and $w>0$ be real numbers.
    \begin{itemize}
        \item \textbf{Product Rule}: $g(uw) = g(u)+g(w)$. In other words, $\log_{b}(uw) = \log_{b}(u) + \log_{b}(w)$.
        \item \textbf{Quotient Rule}: $g(\frac{u}{w}) = g(u)-g(w)$. In other words, $\log_{b}(\frac{u}{w})=\log_{b}(u)-\log_{b}(w)$.
        \item \textbf{Power Rule}: $g(u^w) = wg(u)$. In other words, $\log_{b}(u^w)= w \log_{b}(u)$.
    \end{itemize}
\end{theorem}

\begin{theorem}[Change of Base Formulas]
    Let $a>0, b>0, a \neq 1, b \neq 1$.
    \begin{itemize}
        \item $a^x = b^{x \log_{b}(a)}$ for all real numbers $x$.
        \item $\log_{a}(x) = \frac{\log_{b}(x)}{\log_{b}(a)}$ for all real numbers $x>0$.
    \end{itemize}
\end{theorem}
